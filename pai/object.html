let showIteracje = function(obiekt, paragrafID){
	let output = ""
	for (var wlasciwosc in obiekt){	
		output = output + wlasciwosc + ": " + obiekt[wlasciwosc] + "<br>";
	}
	paragrafID.innerHTML = output;	
}
